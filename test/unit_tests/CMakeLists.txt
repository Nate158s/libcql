set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ".")
set(PROJECT_TEST_NAME ${PROJECT_NAME_STR}_test)

  APPEND PROPERTY COMPILE_FLAGS ${PROJECT_COMPILER_FLAGS})
